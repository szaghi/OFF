00125   pure function stretchinglside(Ns,s_uniform,stretch) result(s_stretch)
00126   !---------------------------------------------------------------------------------------------------------------------------------
00127   !!This function stretches a uniform grid increasing the density of node to the left side of the grid. It is based on the
00128   !!hyperbolic tangent function. The stretched abscissa is computed from the uniform by the equation:
00129   !!\begin{equation}
00130   !!  s_n^{stretch}  = s_L  + \left( {s_R  - s_L } \right)\left[ {1 + \frac{{\tanh \left[ {strecth\left( {\frac{{n - 1}}{{Ns -
00131   !!  1}} - 1} \right)} \right]}}{{\tanh \left( {strecth} \right)}}} \right]\quad for\;n = 1,Ns
00132   !!\end{equation}
00133   !!\noindent where $stretch$ is the intensity of the stretching, $Ns$ is the number of the nodes grid and $s_L,s_R$ are the
00134   !!abscissa value at the left and right bounds.
00135   !---------------------------------------------------------------------------------------------------------------------------------
00136 
00137   !---------------------------------------------------------------------------------------------------------------------------------
00138   implicit none
00139   integer(I4P), intent(IN):: Ns              ! Number of nodes.
00140   real(R_P),    intent(IN):: s_uniform(1:Ns) ! Uniform grid.
00141   real(R_P),    intent(IN):: stretch         ! Intensity of stretch.
00142   real(R_P)::                s_stretch(1:Ns) ! Stretched grid.
00143   real(R_P)::                s_L             ! Left value of abscissa.
00144   real(R_P)::                s_R             ! Right value of abscissa.
00145   integer(I4P)::             n               ! Nodes counter.
00146   !---------------------------------------------------------------------------------------------------------------------------------
00147 
00148   !---------------------------------------------------------------------------------------------------------------------------------
00149   s_L     = s_uniform(1)  ! Calculating the left value of abscissa.
00150   s_R     = s_uniform(Ns) ! Calculating the right value of abscissa.
00151   do n=1,Ns
00152     s_stretch(n)=s_L+(s_R-s_L)*(1._R_P+tanh(stretch*((n-1._R_P)/(Ns-1._R_P)-1._R_P))/tanh(stretch)) ! Stretching the unirform grid.
00153   enddo
00154   return
00155   !---------------------------------------------------------------------------------------------------------------------------------
00156   endfunction stretchinglside

00191   pure function stretching2side(s_sym_in,Ns,s_uniform,Ds_L,Ds_R) result(s_stretch)
00192   !---------------------------------------------------------------------------------------------------------------------------------
00193   !!This function stretching a uniform grid and allowing the imposition of the space steps at the first and at the
00194   !!last cells (2 sides). It is based on the Vinokur (1980) algorithm:
00195   !!\begin{equation}
00196   !!  s_n^{stretch}  = s_L  + \left( {s_R  - s_L } \right)\frac{{u_n }}{{\alpha  + \left( {1 - \alpha } \right)u_n }}\quad
00197   !!  for\;n = 1,Ns
00198   !!\end{equation}
00199   !!\noindent where $u_n$ and $\alpha$ are:
00200   !!\begin{equation}
00201   !!  \begin{array}{l}
00202   !!    u_n  = \frac{{\tanh \left[ {strecth\left( {\frac{{n - 1}}{{Ns - 1}} - s_{sym} } \right)} \right] +
00203   !!           \tanh \left( {strecth \cdot s_{sym} } \right)}}{{\tanh \left[ {strecth\left( {1 - s_{sym} } \right)} \right] +
00204   !!           \tanh \left( {strecth \cdot s_{sym} } \right)}} \\
00205   !!    \alpha  = \sqrt {\frac{{\Delta _R }}{{\Delta _L }}}  \\
00206   !!  \end{array}
00207   !!\end{equation}
00208   !!\noindent where $stretch$ is the intensity of the stretching, $Ns$ is the number of the nodes grid, $s_L,s_R$ are the
00209   !!abscissa value at the left and right bounds and $s_sym$ is the abscissa where the symmetry point is placed; $s_sym$ falls in
00210   !!the range $[0,1]$ where $0$ corrisponds to the left bound and $1$ to right one. Note that the
00211   !!stretch intensity must be calculated according to the space steps imposed at the bounds; the equation for the stretch
00212   !!intensity is:
00213   !!\begin{equation}
00214   !!  \sinh \left( {stretch} \right) - \frac{{stretch}}{{\sqrt {\Delta _L^{adm} \Delta _R^{adm} } }} = 0
00215   !!\end{equation}
00216   !!\noindent where:
00217   !!\begin{equation}
00218   !! \Delta _L^{adm}  = \frac{{\Delta _L }}{{\left( {s_R  - s_L } \right)}}Ns \quad
00219   !! \Delta _R^{adm}  = \frac{{\Delta _R }}{{\left( {s_R  - s_L } \right)}}Ns
00220   !!\end{equation}
00221   !!\noindent are the adimensional left and right space steps. This equation is non linear and must be solverd with an iterative
00222   !!method like Newton-Rapson.
00223   !---------------------------------------------------------------------------------------------------------------------------------
00224 
00225   !---------------------------------------------------------------------------------------------------------------------------------
00226   implicit none
00227   real(R_P),    intent(IN), optional:: s_sym_in              ! Optional input value of abscissa of symmetry.
00228   integer(I4P), intent(IN)::           Ns                    ! Number of nodes.
00229   real(R_P),    intent(IN)::           s_uniform(1:Ns)       ! Uniform grid.
00230   real(R_P),    intent(IN)::           Ds_L                  ! First space step (Left).
00231   real(R_P),    intent(IN)::           Ds_R                  ! Last  space step (Right).
00232   real(R_P)::                          s_sym                 ! Abscissa of symmetry.
00233   real(R_P)::                          s_stretch(1:Ns)       ! Stretched grid.
00234   real(R_P)::                          s_L                   ! Left value of abscissa.
00235   real(R_P)::                          s_R                   ! Right value of abscissa.
00236   real(R_P)::                          c_L                   ! Relative (adimensional) first space step (Left).
00237   real(R_P)::                          c_R                   ! Relative (adimensional) last  space step (Right).
00238   real(R_P)::                          stretch               ! Intensity of stretch.
00239   real(R_P)::                          stretch_n             ! Iterative value of stretch intensity.
00240   real(R_P), parameter::               toll = 10._R_P**(-10) ! Tollerance for computing the stretch intensity.
00241   real(R_P)::                          u_n                   ! Node stretch factor.
00242   integer(I4P)::                       n                     ! Nodes counter.
00243   !---------------------------------------------------------------------------------------------------------------------------------
00244 
00245   !---------------------------------------------------------------------------------------------------------------------------------
00246   if (present(s_sym_in)) then
00247     s_sym = s_sym_in ! Imposing passed value of symmetry abscissa.
00248   else
00249     s_sym = 0.5_R_P  ! Symmetry abscissa is at 50% as default.
00250   endif
00251   s_L       = s_uniform(1)                                             ! Compute left value of abscissa.
00252   s_R       = s_uniform(Ns)                                            ! Compute right value of abscissa.
00253   c_L       = Ds_L/(s_R-s_L)*Ns                                        ! Compute left adm space step.
00254   c_R       = Ds_R/(s_R-s_L)*Ns                                        ! Compute right adm space step.
00255   stretch_n = 1.1_R_P                                                  ! Initialize iterative stretch intensity value.
00256   stretch   = stretch_n-(sinh(stretch_n)-stretch_n/(sqrt(c_L*c_R)))/ &
00257                         (cosh(stretch_n)-1._R_P/(sqrt(c_L*c_R)))       ! Update stretch intensity.
00258   do while(abs(stretch-stretch_n)>toll*abs(stretch))                     ! Evalute iterative convergence.
00259     stretch_n = stretch                                                  ! Update iterative stretch intensity value.
00260     stretch   = stretch_n-(sinh(stretch_n)-stretch_n/(sqrt(c_L*c_R)))/ &
00261                           (cosh(stretch_n)-1._R_P/(sqrt(c_L*c_R)))       ! Update stretch intensity.
00262   enddo
00263   do n=1,Ns
00264     u_n          = (tanh(stretch*((n-1._R_P)/(Ns-1._R_P)-s_sym)) + tanh(stretch*s_sym))/ &
00265                    (tanh(stretch*(1._R_P-s_sym))                 + tanh(stretch*s_sym))    ! Computing u_n.
00266     s_stretch(n) = s_L+(s_R-s_L)*u_n/(sqrt(c_R/c_L) + (1._R_P-sqrt(c_R/c_L))*u_n)          ! Stretching the unirform grid.
00267   enddo
00268   return
00269   !---------------------------------------------------------------------------------------------------------------------------------
00270   endfunction stretching2side
